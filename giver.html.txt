  <section id="my-items" class="grid grid-cols-1 gap-4"></section>

  <script type="module" src="../firebase-init.js"></script>
  <script type="module">
    import { auth } from "../firebase-init.js";
    import { onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-auth.js";
    import { db } from "../firebase-init.js";
    import { collection, query, where, onSnapshot, orderBy } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-firestore.js";

    const myItemsContainer = document.getElementById("my-items");
    onAuthStateChanged(auth, user => {
      if (!user) location.replace("/index.html");
      else {
        const q = query(collection(db, "items"), where("ownerId", "==", user.uid), orderBy("createdAt","desc"));
        onSnapshot(q, snap => {
          myItemsContainer.innerHTML = "";
          snap.docs.forEach(d => {
            const data = d.data();
            const el = document.createElement("div");
            el.className = "card";
            el.innerHTML = `
              <div class="flex gap-4">
                <img src="${data.imageURL || ''}" alt="" style="width:80px;height:80px;object-fit:cover;border-radius:6px"/>
                <div class="flex-1">
                  <h3 class="font-semibold">${data.title}</h3>
                  <p class="text-sm">${data.description || ''}</p>
                  <p class="text-xs text-gray-500">Status: ${data.status}</p>
                </div>
              </div>`;
            myItemsContainer.appendChild(el);
          });
        });
      }
    });

    document.getElementById("logout").addEventListener("click", async () => {
      await signOut(auth);
      location.replace("/index.html");
    });
  </script>
